Arch install/config steps

Boot fresh VM
ip link
systemctl restart dhcpcd.service
timedatectl set-ntp true
timedatectl status      # sanity check
lsblk
parted /dev/sda
mklabel msdos
mkpart primary ext4 1MiB 100%
quit
lsblk /dev/sda
mkfs.ext4 /dev/sda1
mount /dev/sda1 /mnt
pacstrap -i /mnt base base-devel
genfstab -U /mnt > /mnt/etc/fstab
cat /mnt/etc/fstab
arch-chroot /mnt /bin/bash
nano /etc/locale.gen			# uncomment en_US.UTF-8 UTF-8
locale-gen
echo LANG=en_US.UTF-8 > /etc/locale.conf
ln -sf /usr/share/zoneinfo/America/New_York /etc/localtime
hwclock --systohc --utc
pacman -S grub os-prober
grub-install --recheck /dev/sda
grub-mkconfig -o /boot/grub/grub.cfg
echo workstation > /etc/hostname	# choose a hostname for your vm
nano /etc/hosts						# add hostname after localhost
systemctl enable dhcpcd@eth0.service
passwd								# set root password
useradd -m -g users -G wheel,storage,power -s /bin/bash [username]
passwd [username]
pacman -S sudo
EDITOR=nano visudo					# uncomment %wheel ALL=(ALL) ALL
exit
umount -R /mnt
reboot
login as non-root user
sudo dhcpcd
sudo pacman -S alsa-utils
sudo pacman -S xorg-server xorg-xinit xorg-server-utils mesa ttf-dejavu samba smbclient networkmanager 
    networkmanager-vpnc networkmanager-pptp networkmanager-openconnect network-manager-applet 
    gvfs gvfs-smb sshfs git go bspwm sxhkd dmenu
# previous installs x window system, fonts, network stuff > 500MB installed
sudo systemctl enable NetworkManager
lspci -k | grep -A 2 -i "VGA" 					# show graphics card sanity check
sudo pacman -S virtualbox-guest-utils			# for virtualbox only
sudo pacman -S xf86-video-vmware xf85-input-vmmouse # for vmware only
curl https://raw.githubusercontent.com/Djent-/arch-config-2015/2016/.xinitrc > .xinitrc
curl https://raw.githubusercontent.com/Djent-/arch-config-2015/2016/.Xresources > .Xresources
xrdb -merge ./.Xresources
mkdir .config
mkdir .config/bspwm
mkdir .config/sxhkd
curl https://raw.githubusercontent.com/Djent-/arch-config-2015/2016/bspwmrc > bspwmrc						# add bspwm config
curl https://raw.githubusercontent.com/Djent-/arch-config-2015/2016/sxhkdrc > bspwmrc						# add sxhkd config
sudo pacman -S firefox rxvt-unicode zsh ranger
sudo pacman -S mpd mpc ncmpcpp			# music player + etc
echo $SHELL
sh -c "$(curl -fsSL https://raw.githubusercontent.com/robbyrussell/oh-my-zsh/master/tools/install.sh)"
mkdir build-repos
mkdir build-tars
mkdir .nano
curl https://raw.githubusercontent.com/Djent-/arch-config-2015/2016/.nanorc > .nanorc
curl https://raw.githubusercontent.com/Djent-/arch-config-2015/2016/go.nanorc > .nano/go.nanorc
go get github.com/nsf/termbox-go
go install github.com/djent-/go-diags

Install stylish and Twily's powerline firefox
install scite and custom theme
Setup GOPATH and add it to PATH
